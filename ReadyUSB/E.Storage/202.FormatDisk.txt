STRING Get-Disk|Select Number,FriendlyName,Size,PartitionStyle,IsSystem,IsBoot|Format-Table -AutoSize;$sel=Read-Host "Enter disk NUMBERS to format (comma-separated, e.g. 0,1,2)";$nums=($sel -split ',')|%{$_.Trim()}|?{$_ -match '^\d+$'}|%{[int]$_};$safe=@();$warn=@();$skipped=@();foreach($n in $nums){$d=Get-Disk -Number $n -ErrorAction SilentlyContinue;if(-not $d){$skipped+=("Disk ${n}: not found");continue};$hasC=Get-Partition -DiskNumber $n -ErrorAction SilentlyContinue|Get-Volume -ErrorAction SilentlyContinue|?{$_.DriveLetter -eq 'C'};if($d.IsSystem -or $d.IsBoot -or $hasC){$skipped+=("Disk ${n}: SYSTEM/BOOT/C: — blocked");continue};$flags=Get-Partition -DiskNumber $n -ErrorAction SilentlyContinue|?{$_.IsBoot -or $_.IsSystem -or $_.IsRecovery};if($flags){$warn+=$n}else{$safe+=$n}};if(-not $safe -and -not $warn){Write-Host "No eligible disks selected.";if($skipped){$skipped|%{Write-Host $_}};return};if($warn){Write-Host ("Needs override (EFI/Recovery present): {0}" -f ($warn -join ', ')) -ForegroundColor Yellow};if($safe){Write-Host ("Will format (safe): {0}" -f ($safe -join ', ')) -ForegroundColor Cyan};$final=$safe+@();if($warn){$exp="OVERRIDE {0}" -f ($warn -join ',');$ok=Read-Host ("Type '{0}' to ALSO wipe EFI/Recovery disks" -f $exp);if($ok -eq $exp){$final+=$warn}};if(-not $final){Write-Host "Aborted.";if($skipped){$skipped|%{Write-Host $_}};return};$expect="FORMAT {0}" -f ($final -join ',');$confirm=Read-Host ("Type '{0}' to proceed" -f $expect);if($confirm -ne $expect){Write-Host "Aborted.";return};foreach($n in $final){try{Get-Partition -DiskNumber $n -ErrorAction SilentlyContinue|Remove-Partition -Confirm:$false -ErrorAction SilentlyContinue;Clear-Disk -Number $n -RemoveData -RemoveOEM -Confirm:$false;Initialize-Disk -Number $n -PartitionStyle GPT;$p=New-Partition -DiskNumber $n -UseMaximumSize -AssignDriveLetter;Format-Volume -Partition $p -FileSystem NTFS -NewFileSystemLabel ("RAIDTestDisk{0}" -f $n) -Confirm:$false -Force;Write-Host ("Disk {0} formatted successfully." -f $n)}catch{Write-Host ("Disk {0} error: {1}" -f $n,$_.Exception.Message)}};if($skipped){Write-Host ("Skipped:`n - {0}" -f ($skipped -join "`n - "))}
ENTER
