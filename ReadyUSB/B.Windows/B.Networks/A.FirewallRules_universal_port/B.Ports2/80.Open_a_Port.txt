REM Verificar, eliminar y abrir un puerto en el firewall de Windows (interactivo)
DELAY 1500
STRING $port=Read-Host "Enter the port (e.g. 3389)"; Get-NetFirewallRule | Where-Object { $_.Action -eq 'Block' -and (Get-NetFirewallPortFilter -AssociatedNetFirewallRule $_ -ErrorAction SilentlyContinue | ForEach-Object {$_.LocalPort}) -contains $port } | Remove-NetFirewallRule; New-NetFirewallRule -DisplayName "Allow Port $port (TCP-In)" -Direction Inbound -Action Allow -Protocol TCP -LocalPort $port -Profile Any -ErrorAction SilentlyContinue | Out-Null; New-NetFirewallRule -DisplayName "Allow Port $port (UDP-In)" -Direction Inbound -Action Allow -Protocol UDP -LocalPort $port -Profile Any -ErrorAction SilentlyContinue | Out-Null; Get-NetFirewallRule | Where-Object { (Get-NetFirewallPortFilter -AssociatedNetFirewallRule $_ -ErrorAction SilentlyContinue | ForEach-Object {$_.LocalPort}) -contains $port } | Select-Object DisplayName,Action,Direction,Profile,Enabled | Format-Table -Auto
ENTER
