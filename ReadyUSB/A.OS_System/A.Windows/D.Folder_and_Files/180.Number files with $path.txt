REM ReadyUSB - Rename files in folder with numerical prefix
DELAY 1500
STRING $ruta=Read-Host "Enter the folder path to rename files"; $files=Get-ChildItem -Path $ruta -File | Sort-Object Name; $counter=1; foreach ($file in $files) { $newName="$counter.$($file.Name)"; Rename-Item -Path $file.FullName -NewName $newName; $counter++ }; Write-Host "✅ Files have been renamed with a numerical prefix."
ENTER
