REM Create a shared folder (interactivo ReadyUSB)
DELAY 1500
STRING if (-not ([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltinRole]::Administrator)) { Start-Process powershell -ArgumentList "-NoProfile -ExecutionPolicy Bypass -File `"$PSCommandPath`"" -Verb runAs; exit }; $path = Read-Host "Enter folder path to share (e.g. D:\Apps)"; if (-not (Test-Path -LiteralPath $path)) { Write-Host "Folder not found"; exit }; $shareNameDefault = Split-Path $path -Leaf; $shareName = Read-Host "Share name (Enter for $shareNameDefault)"; if ([string]::IsNullOrWhiteSpace($shareName)) { $shareName = $shareNameDefault }; $mode = Read-Host "Press 1 for existing user, 2 to create a new local user"; if ($mode -eq '2') { $newUser = Read-Host "Enter NEW local username"; if ([string]::IsNullOrWhiteSpace($newUser)) { Write-Host "Error: empty username"; exit }; try { if (Get-LocalUser -Name $newUser -ErrorAction Stop) { Write-Host "Error: user already exists"; exit } } catch {}; $p1 = Read-Host "Enter password" -AsSecureString; $p2 = Read-Host "Confirm password" -AsSecureString; if (([PSCredential]::new('u',$p1).GetNetworkCredential().Password) -ne ([PSCredential]::new('u',$p2).GetNetworkCredential().Password)) { Write-Host "Error: passwords do not match"; exit }; New-LocalUser -Name $newUser -Password $p1 -PasswordNeverExpires:$true -AccountNeverExpires:$true | Out-Null; try { Add-LocalGroupMember -Group "Users" -Member $newUser -ErrorAction SilentlyContinue } catch {}; $account = "$env:COMPUTERNAME\$newUser" } elseif ($mode -eq '1') { $inputName = Read-Host "Enter EXISTING username (DOMAIN\User or COMPUTER\User)"; if ([string]::IsNullOrWhiteSpace($inputName)) { Write-Host "Error: empty username"; exit }; try { $null = ([System.Security.Principal.NTAccount]$inputName).Translate([System.Security.Principal.SecurityIdentifier]).Value } catch { Write-Host "Error: name not found"; exit }; $account = $inputName } else { Write-Host "Error: choose 1 or 2"; exit }; Set-Service -Name LanmanServer -StartupType Automatic; Start-Service -Name LanmanServer; $fps = Get-NetFirewallRule -Name 'FPS-SMB-In-TCP','FPS-SMB-In-UDP' -ErrorAction SilentlyContinue; if ($fps) { Enable-NetFirewallRule -Name ($fps | Select-Object -ExpandProperty Name) | Out-Null } else { $c = Get-NetFirewallRule -ErrorAction SilentlyContinue | Where-Object { $_.DisplayGroup -match 'File and Printer|Uso compartido de archivos' }; if ($c) { $c | Enable-NetFirewallRule | Out-Null } }; $existing = Get-SmbShare -Name $shareName -ErrorAction SilentlyContinue; if ($existing) { try { Grant-SmbShareAccess -Name $shareName -AccountName $account -AccessRight Full -Force | Out-Null } catch {} } else { New-SmbShare -Name $shareName -Path $path -CachingMode None -FullAccess $account | Out-Null }; $keepEveryone = Read-Host "Grant Everyone READ on the share? (Y/N, default Y)"; if ([string]::IsNullOrWhiteSpace($keepEveryone) -or $keepEveryone.ToUpper() -eq 'Y') { try { Grant-SmbShareAccess -Name $shareName -AccountName 'Everyone' -AccessRight Read -Force | Out-Null } catch {} }; $acl = Get-Acl -LiteralPath $path; $rule = New-Object System.Security.AccessControl.FileSystemAccessRule($account,"Modify,Synchronize","ContainerInherit, ObjectInherit","None","Allow"); $acl.SetAccessRule($rule); Set-Acl -LiteralPath $path -AclObject $acl; $unc = "\\$($env:COMPUTERNAME)\$shareName"; Write-Host "Done. Share: $unc"; Write-Host "Granted to: $account (Share Full, NTFS Modify)"
ENTER
