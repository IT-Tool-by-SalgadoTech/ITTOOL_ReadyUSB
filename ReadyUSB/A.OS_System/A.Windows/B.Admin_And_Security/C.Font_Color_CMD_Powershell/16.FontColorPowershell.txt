function Upsert-Prop([object]$obj,[string]$name,$value){ if($obj.PSObject.Properties.Match($name).Count -eq 0){ Add-Member -InputObject $obj -NotePropertyName $name -NotePropertyValue $value -Force } else { $obj.$name=$value } };
