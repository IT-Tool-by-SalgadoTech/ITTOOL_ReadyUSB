REM Ejecutar netstat -ab desde PowerShell como administrador
DELAY 1000
STRING reg add "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server" /v fDenyTSConnections /t REG_DWORD /d 1 /f >$null; Disable-NetFirewallRule -DisplayGroup "Remote Desktop" -ea SilentlyContinue; Get-NetFirewallRule | ?{ $_.Action -eq 'Allow' -and (Get-NetFirewallPortFilter -AssociatedNetFirewallRule $_ -ea SilentlyContinue | %{$_.LocalPort}) -contains 3389 } | Remove-NetFirewallRule -ea SilentlyContinue; Get-NetFirewallRule -DisplayName "Block Port 3389*" -ea SilentlyContinue | Remove-NetFirewallRule -ea SilentlyContinue; New-NetFirewallRule -DisplayName "Block Port 3389 (TCP-In)" -Direction Inbound -Action Block -Protocol TCP -LocalPort 3389 -Profile Any -Enabled True -ea SilentlyContinue | Out-Null; New-NetFirewallRule -DisplayName "Block Port 3389 (UDP-In)" -Direction Inbound -Action Block -Protocol UDP -LocalPort 3389 -Profile Any -Enabled True -ea SilentlyContinue | Out-Null; New-NetFirewallRule -DisplayName "Block Port 3389 (All-In)" -Direction Inbound -Action Block -Protocol Any -LocalPort 3389 -Profile Any -Enabled True -ea SilentlyContinue | Out-Null; Stop-Service -Name TermService -Force -ea SilentlyContinue; Set-Service -Name TermService -StartupType Manual; Get-NetFirewallRule | ?{ (Get-NetFirewallPortFilter -AssociatedNetFirewallRule $_ -ea SilentlyContinue | %{$_.LocalPort}) -contains 3389 } | Select DisplayName,Action,Direction,Profile,Enabled | ft -Auto
ENTER
