STRING $IsAdmin=([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator);if(-not $IsAdmin){Write-Error "Run PowerShell as Administrator";exit 1};Write-Output "Report generated: $(Get-Date -Format yyyy-MM-dd_HH:mm:ss)";Write-Output "=== Local Users ===";try{$localUsers=Get-LocalUser -ErrorAction Stop | Select-Object Name,Enabled,Description,SID}catch{$localUsers=Get-CimInstance Win32_UserAccount -Filter 'LocalAccount=True' | Select-Object Name,@{Name='Enabled';Expression={-not$_.Disabled}},Description,SID};$localUsers|Format-Table -AutoSize;Write-Output "=== Administrators group members ===";try{$admins=Get-LocalGroupMember -Group Administrators -ErrorAction Stop | Select-Object @{Name='Name';Expression={$_.Name}},@{Name='ObjectType';Expression={$_.ObjectClass}},@{Name='SID';Expression={$_.SID.Value}}}catch{$group=[ADSI]"WinNT://$env:COMPUTERNAME/Administrators,group";$members=@();$group.Members()|ForEach-Object{$n=$_.GetType().InvokeMember('Name','GetProperty',$null,$_,$null);$c=$_.GetType().InvokeMember('Class','GetProperty',$null,$_,$null);$sidObj=Get-CimInstance Win32_UserAccount -Filter "Name='$n' AND LocalAccount=True" -ErrorAction SilentlyContinue;$sid=if($sidObj){$sidObj.SID}else{$null};$members+=[PSCustomObject]@{Name=$n;ObjectType=$c;SID=$sid}};$admins=$members};if($admins){$admins|Format-Table -AutoSize}else{Write-Output "No administrators found"};Write-Output "=== Domain / Workgroup Info ===";$cs=Get-CimInstance Win32_ComputerSystem;if($cs.PartOfDomain){Write-Output "Domain: $($cs.Domain)"}else{Write-Output "Workgroup: $($cs.Domain)"};Write-Output "=== Summary ===";Write-Output "Local users: $($localUsers.Count)";Write-Output "Administrators members: $($admins.Count)";Write-Output "Domain joined: $($cs.PartOfDomain)"
