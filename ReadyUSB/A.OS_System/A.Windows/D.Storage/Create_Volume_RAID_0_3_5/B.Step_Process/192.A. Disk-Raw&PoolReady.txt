DELAY 1800
STRING $nums=(Read-Host "Enter disk numbers to PREP (e.g. 0,1,2)").Split(',')|%{$_.Trim()}|?{$_ -match '^\d+$'}|%{[int]$_};if(-not $nums){Write-Host "No disk numbers provided. Exiting.";return};foreach($n in $nums){try{Write-Host "Preparing Disk $n ...";Set-Disk -Number $n -IsReadOnly $false -ErrorAction SilentlyContinue;Set-Disk -Number $n -IsOffline $false -ErrorAction SilentlyContinue;Clear-Disk -Number $n -RemoveData -RemoveOEM -Confirm:$false -ErrorAction Stop;Write-Host "Disk $n cleared to RAW."}catch{Write-Host "Warning prepping disk $n : $($_.Exception.Message)"}};$phys=foreach($n in $nums){Get-PhysicalDisk|Where-Object{$_.DeviceId -eq $n}};if($phys){$phys|Reset-PhysicalDisk -ErrorAction SilentlyContinue;$phys|Set-PhysicalDisk -Usage AutoSelect -ErrorAction SilentlyContinue};Update-HostStorageCache;Write-Host "`nSelected disks status:" -ForegroundColor Cyan;Get-PhysicalDisk|Where-Object{$nums -contains $_.DeviceId}|Select DeviceId,FriendlyName,CanPool,Usage,Size,BusType,OperationalStatus|Format-Table -AutoSize
ENTER
