REM Enumerate web directories and files with gobuster
DELAY 500
STRING read -p "Target URL (e.g. http://192.168.1.1): " u; read -p "Wordlist (Enter for default): " w; w="${w:-/usr/share/wordlists/dirb/common.txt}"; gobuster dir -u "$u" -w "$w" 2>/dev/null || echo "gobuster not installed. Run: sudo apt install gobuster"
ENTER
