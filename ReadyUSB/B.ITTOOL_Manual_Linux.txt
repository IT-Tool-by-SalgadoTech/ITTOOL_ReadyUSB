#NOTE IT-Tool - Welcome to ReadyUSB (Linux)
#NOTE Downloads Manual_and_Welcome.sh from GitHub and runs it
WTIME 1800
SCRIPT U="${SUDO_USER:-$USER}"; H=$(getent passwd "$U"|cut -d: -f6); D="$H/ITTOOL"; F="$D/IT-Tool_Manual.pdf"; L="https://github.com/IT-Tool-by-SalgadoTech/ittool-External_Tools/raw/refs/heads/main/IT-Tool%20Manual.pdf"; mkdir -p "$D"; command -v curl >/dev/null 2>&1 || command -v wget >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y curl; command -v dnf >/dev/null 2>&1 && dnf install -y curl; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm curl; }; if command -v curl >/dev/null 2>&1; then curl -fsSL "$L" -o "$F"; else wget -qO "$F" "$L"; fi; chown "$U": "$D" "$F" 2>/dev/null; xdg-open "$F" >/dev/null 2>&1
ENTER
