#NOTE IT-Tool - Launch Script_Saver (Linux)
#NOTE Downloads Script_Saver_Linux.sh to ITTOOL in user home, 3-distro safe, runs it interactively
WTIME 800
SCRIPT U="${SUDO_USER:-$USER}"; H=$(getent passwd "$U"|cut -d: -f6); D="$H/ITTOOL"; F="$D/Script_Saver_Linux.sh"; L="https://github.com/IT-Tool-by-SalgadoTech/ReaduUSB_With_PS1/raw/refs/heads/main/ReadyUSB/B.Script_Saver_Runner/Script_Saver_Linux.sh"; mkdir -p "$D"; command -v curl >/dev/null 2>&1 || command -v wget >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y curl; command -v dnf >/dev/null 2>&1 && dnf install -y curl; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm curl; }; if command -v curl >/dev/null 2>&1; then curl -fsSL "$L" -o "$F"; else wget -qO "$F" "$L"; fi; chmod +x "$F"; chown "$U": "$D" "$F" 2>/dev/null; bash "$F" < /dev/tty
ENTER
