#NOTE Installs deps (Debian/Fedora/Arch), downloads Bluetooth_Off.sh (+ _itlib.sh) to ~/ITTOOL, runs it
WTIME 800
SCRIPT U="${SUDO_USER:-$USER}"; H=$(getent passwd "$U"|cut -d: -f6); D="$H/ITTOOL"; F="$D/Bluetooth_Off.sh"; L="https://github.com/IT-Tool-by-SalgadoTech/ReaduUSB_With_PS1/raw/refs/heads/main/ReadyUSB/D.IT_Angel/Fast_Defense/Linux/Bluetooth_Off.sh"; LIB="$D/_itlib.sh"; LL="https://github.com/IT-Tool-by-SalgadoTech/ReaduUSB_With_PS1/raw/refs/heads/main/ReadyUSB/D.IT_Angel/Fast_Defense/Linux/_itlib.sh"; mkdir -p "$D"; command -v curl >/dev/null 2>&1 || command -v wget >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y curl; command -v dnf >/dev/null 2>&1 && dnf install -y curl; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm curl; }; if command -v curl >/dev/null 2>&1; then curl -fsSL "$L" -o "$F"; curl -fsSL "$LL" -o "$LIB"; else wget -qO "$F" "$L"; wget -qO "$LIB" "$LL"; fi; chmod +x "$F"; chown "$U": "$D" "$F" "$LIB" 2>/dev/null; command -v rfkill >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y rfkill; command -v dnf >/dev/null 2>&1 && dnf install -y util-linux; command -v yum >/dev/null 2>&1 && yum install -y util-linux; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm util-linux; }; command -v bluetoothctl >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y bluez; command -v dnf >/dev/null 2>&1 && dnf install -y bluez; command -v yum >/dev/null 2>&1 && yum install -y bluez; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm bluez; }; bash "$F"
ENTER
