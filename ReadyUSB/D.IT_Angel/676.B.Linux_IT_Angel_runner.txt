SCRIPT U="${SUDO_USER:-$USER}"; H=$(getent passwd "$U"|cut -d: -f6); D="$H/ITTOOL"; F="$D/IT_Angel.py"; V="$D/venv"; P="$V/bin/python"; L="https://github.com/IT-Tool-by-SalgadoTech/ReaduUSB_With_PS1/raw/refs/heads/main/ReadyUSB/D.IT_Angel/IT_Angel.py"; G="https://bootstrap.pypa.io/get-pip.py"; mkdir -p "$D"; command -v curl >/dev/null 2>&1 || command -v wget >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y curl; command -v dnf >/dev/null 2>&1 && dnf install -y curl; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm curl; }; command -v python3 >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y python3; command -v dnf >/dev/null 2>&1 && dnf install -y python3; command -v pacman >/dev/null 2>&1 && pacman -Sy --noconfirm python; }; python3 -c 'import ensurepip' >/dev/null 2>&1 || { command -v apt-get >/dev/null 2>&1 && apt-get install -y python3-venv python3-pip; }; python3 -m venv "$V" 2>/dev/null || python3 -m venv --without-pip "$V"; "$P" -m ensurepip --upgrade 2>/dev/null; "$P" -m pip --version >/dev/null 2>&1 || { if command -v curl >/dev/null 2>&1; then curl -fsSL "$G" -o /tmp/get-pip.py; else wget -qO /tmp/get-pip.py "$G"; fi; "$P" /tmp/get-pip.py; }; "$P" -m pip install --upgrade pip setuptools wheel >/dev/null 2>&1; "$P" -m pip install psutil openpyxl colorama rich textual pyserial pyusb; if command -v curl >/dev/null 2>&1; then curl -fsSL "$L" -o "$F"; else wget -qO "$F" "$L"; fi; chown -R "$U": "$D" 2>/dev/null; "$P" "$F"
